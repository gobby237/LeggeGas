\section{Obiettivo}
Verifica della legge dei gas ideali per l'aria
\begin{equation}
    PV = nRT
\end{equation}

I due grafici presentano caratteristiche comuni, che necessitano spiegazione. Si nota come nelle code delle linee possa sembrare che la diminuzione di pressione non sia seguita dal relativo aumento di volume, ciò è dato dal fatto che lo strumento a fine corsa smette di rilevare la variazione dell'ultimo, in quanto non costruito per lavorare a pressioni troppo basse. Nell'analisi dati ci si è limitati a non considerare quell'ultimo tratto. 
\[\]
Inoltre sono evidenti altre parti in cui la linea devia dalla sua traiettoria rettilinea; ovvero con volume a poco meno di 9 $ \mathrm{cm}^3$, e circa 11 $\mathrm{cm}^3$. Lì si vede un piccolo scalino in cui la pressione sembra variare senza che vari il volume, dal momento che si possono trovare in entrambi i grafici sia in compressione che dilatazione, deduciamo che anche qui si tratta di un problema dello strumento; e siccome gli errori si sviluppano orizzontalmente, sempre allo stesso volume, arriviamo alla conclusione che sia un problema del sensone che tiene conto di questa grandezza termodinamica. Piccole imperfezioni o possibili granelli di polvere, che compromettono momentaneamente la corretta raccolta dati dello strumento.

\noindent\textbf{Legenda formule errori relativi (riferimento colonna):}
\begin{itemize}
    \item Errore casuale su \( b \): \( \pm \frac{s_b}{b} \)
    \item Errori di taratura su \( V \) e \( p \): \( \pm \frac{2\delta V}{\sqrt{3}\Delta V}, \quad \pm \frac{2\delta p}{\sqrt{3}\Delta p} \)
    \item Non quasi staticità: \( \pm \frac{|n_{\text{dilat}} - n_{\text{comp}}|}{2\sqrt{3}\bar{n}} \)
    \item Non isoterma: \( \pm \frac{s_T}{T} \)
    \item Errore di taratura su \( T \): \( \pm \frac{\delta_T}{\sqrt{3}T} \)
    \item Errore sistema non chiuso: \( \frac{\delta n_j}{n_j} = -\frac{\alpha}{n_j} \left( TM_j + \frac{tm_j}{2} \right); \quad \pm \frac{\delta n_1 - \delta n_{\text{ultimo}}}{2\sqrt{3}\bar{n}} \)
\end{itemize}

\subsection{Formule}
\vspace{0.2cm}

\begin{itemize}

\item \textbf{Media aritmetica}
\begin{equation}
    \bar{x} = \frac{1}{N} \sum_{i=1}^{N} x_i
\end{equation}

\item \textbf{Deviazione standard della singola misura} 

    \begin{equation}
        \sigma = \sqrt{\frac{1}{N-1} \sum_{i=1}^{N} (x_i - \bar{x})^2}
    \end{equation} 
\item \textbf{Deviazione standard della media per misure statisticamente indipendenti}
\begin{equation}
    \sigma_{\bar{x}} = \frac{\sigma}{\sqrt{N}}
\end{equation}

\item \textbf{Errore relativo}
\begin{equation}
    \varepsilon_r = \frac{\sigma}{x}
\end{equation}
 \item \textbf{Interpolazione lineare \( y = a + bx \) con} 
    \begin{itemize}
        \item \( \forall i \) \( x_i \) statisticamente indipendenti tra loro, \( y_i \) statisticamente indipendenti tra loro;
        \item \( \forall i \) \( \sigma_{x}/x \ll \sigma_{y}/y \), e \( \sigma_i = \sigma_y \)
    \end{itemize}
    
    \begin{equation}
        a_i = \frac{1}{\Delta} \left( \sum_i x_i^2 - x_i \sum_i x_j \right)
    \end{equation}

    \begin{equation}
        b_i = \frac{1}{\Delta} \left( N \sum_i x_i - \sum_i x_j \right)
    \end{equation}

    \begin{equation}
        \sigma_a = \sigma_y \sqrt{ \frac{\sum_i x_i^2}{\Delta} }
    \end{equation}

    \begin{equation}
        \sigma_b = \sigma_y \sqrt{ \frac{N}{\Delta} }
    \end{equation}
 
\begin{equation}
    \Delta = N \sum_i x_i^2 - \left( \sum_i x_i \right)^2
\end{equation}

    \begin{equation}
        \sigma_{y,\text{post}} = \sqrt{ \frac{\sum_{i=1}^{N} (y_i - a - bx_i)^2}{N-2} }
    \end{equation}

\end{itemize}
